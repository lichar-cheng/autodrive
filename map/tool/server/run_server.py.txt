import asyncio
import logging

import uvicorn

from app.config import CONFIG


logging.basicConfig(level=logging.INFO, format="%(asctime)s | %(levelname)s | %(name)s | %(message)s")
logger = logging.getLogger("autodrive.server.launcher")


def main() -> None:
    config = uvicorn.Config("app.main:app", host=CONFIG.host, port=CONFIG.port, reload=False, log_level="info")
    server = uvicorn.Server(config)
    try:
        asyncio.run(server.serve())
    except KeyboardInterrupt:
        logger.info("KeyboardInterrupt received, shutting down server")


if __name__ == "__main__":
    main()
